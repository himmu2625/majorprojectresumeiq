into structured text. This is the most fragile stage due to PDF format diversity and must handle edge cases gracefully.# Core parsing logicimport pdfplumberimport spacydef extract_resume_text(pdf_bytes: bytes) -> dict:    with pdfplumber.open(io.BytesIO(pdf_bytes)) as pdf:        full_text = ''        for page in pdf.pages:            full_text += page.extract_text(x_tolerance=3, y_tolerance=3) or ''        # Section segmentation using spaCy + keyword rules    nlp = spacy.load('en_core_web_sm')    sections = segment_sections(full_text, nlp)    return sections  # {'skills': str, 'experience': str, 'education': str, 'summary': str}Section Keyword TriggersExample Phrases DetectedSkills SectionSkills, Technical Skills, Core Competencies, Technologies, ToolsExperience SectionExperience, Work History, Employment, Professional BackgroundEducation SectionEducation, Academic Background, Qualifications, DegreesSummary SectionSummary, Objective, Profile, About Me, Career Objective4.3  Stage 2 — SBERT Embedding GenerationSentence-BERT converts text segments into 384-dimensional dense vectors that capture semantic meaning. Two texts with similar meaning will have vectors with high cosine similarity, even if they use different words.from sentence_transformers import SentenceTransformer, utilMODEL_NAME = 'all-MiniLM-L6-v2'  # 80MB, 384 dimensionsmodel = SentenceTransformer(MODEL_NAME)  # Load once at startupdef generate_embeddings(resume_sections: dict, job_description: str) -> dict:    # Full document embeddings    resume_full_text = ' '.join(resume_sections.values())    resume_embedding = model.encode(resume_full_text, convert_to_tensor=True)    jd_embedding     = model.encode(job_description, convert_to_tensor=True)        # Section-level embeddings    section_embeddings = {        section: model.encode(text, convert_to_tensor=True)        for section, text in resume_sections.items() if text    }    return { 'full': (resume_embedding, jd_embedding),             'sections': section_embeddings }4.4  Stage 3 — Similarity Scoringfrom sklearn.metrics.pairwise import cosine_similarityimport numpy as npdef compute_scores(embeddings: dict, jd_embedding) -> dict:    # Overall score    r_vec, jd_vec = embeddings['full']    overall = float(util.cos_sim(r_vec, jd_vec)[0][0])    overall_score = round(overall * 100, 1)  # Normalize to 0-100        # Section scores    section_scores = {}    for section, s_vec in embeddings['sections'].items():        sim = float(util.cos_sim(s_vec, jd_vec)[0][0])        section_scores[section] = round(sim * 100, 1)        return { 'overall_score': overall_score,             'section_scores': section_scores,             'confidence': get_confidence(overall_score) }def get_confidence(score: float) -> str:    if score >= 70: return 'High'    elif score >= 45: return 'Medium'    else: return 'Low'4.5  Stage 4 — SHAP Explainability LayerSHAP (SHapley Additive exPlanations) computes the contribution of each input token/phrase to the final similarity score. This is the core differentiator — it makes the black-box model transparent.import shapfrom sklearn.pipeline import Pipelinefrom sklearn.feature_extraction.text import TfidfVectorizerfrom sklearn.linear_model import Ridgedef generate_xai_explanation(resume_text: str, jd_text: str,                              score: float, extracted_skills: list) -> dict:    # Build TF-IDF based explainer (SHAP works on feature vectors)    vectorizer = TfidfVectorizer(ngram_range=(1,3), max_features=500)    combined   = [resume_text, jd_text]    tfidf_matrix = vectorizer.fit_transform(combined)    feature_names = vectorizer.get_feature_names_out()        # Identify matched and missing terms    resume_terms = set(vectorizer.transform([resume_text]).nonzero()[1])    jd_terms     = set(vectorizer.transform([jd_text]).nonzero()[1])        matched = [feature_names[i] for i in resume_terms & jd_terms][:5]    missing = [feature_names[i] for i in jd_terms - resume_terms][:5]        explanation = generate_explanation_text(score, matched, missing)    return {        'positive_phrases': matched,        'missing_keywords': missing,        'explanation_text': explanation,        'improvement_suggestions': generate_suggestions(missing, score)    }4.6  ML Model ConfigurationParameterValue / SettingModel nameall-MiniLM-L6-v2Embedding dimensions384Max sequence length256 tokensSimilarity metricCosine similarityScore normalizationMultiply by 100, round to 1 decimalN-gram range for XAI1-gram to 3-gram (unigrams + bigrams + trigrams)Max TF-IDF features500 (covers typical JD vocabulary)Positive phrases returnedTop 5 (configurable via env var)Missing keywords returnedTop 5 (configurable via env var)Model load strategySingleton — loaded once at FastAPI startupInference timeout15 seconds (returns 504 if exceeded)Batch processingSequential with async queue; 10 resumes max per batch5.
